# /etc/nginx/sites-available/stream-overlay
# Nginx Konfiguration für das Stream Overlay System mit SSL

server {
    # Port 80 wird auf HTTPS umgeleitet
    listen 80;
    server_name floppy5004.net;

    # Certbot-Validierung erlauben
    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    # Port 443 für HTTPS
    listen 443 ssl http2;
    server_name floppy5004.net;

    # SSL Zertifikate von Let's Encrypt
    ssl_certificate /etc/letsencrypt/live/floppy5004.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/floppy5004.net/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Haupt-Proxy zur Node.js-Anwendung
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        
        # Wichtig für WebSockets
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        
        # Weitere wichtige Header
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
